package xyz;
class Bankcustomer{
	int acno;
	String actype;
	int amt;
	Bankcustomer(){
		acno=1;
		actype="saving";
		amt=1000;
	}
	
	Bankcustomer(int x,String y,int z){
		acno=x;
		actype=y;
		amt=z;
	}
	void show2() {
		System.out.println(acno+" "+actype+" "+amt);

	}
}

public class Lab3q2 {

	public static void main(String[] args) {
		Bankcustomer b1 =new Bankcustomer();
		b1.show2();

		b2.show2();
